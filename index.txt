    <script>
        // Typing Animation Text
        const quote = "Through all of life's complex waves, ups, and downs,\nthe final summation is always 'Us'.";
        const typingSpeed = 50;
        let charIndex = 0;

        const music = document.getElementById("bgMusic");
        const musicBtnText = document.getElementById("musicBtnText");
        const musicIcon = document.getElementById("musicIcon");

        function toggleMusic() {
            if (music.paused) {
                music.play().then(() => {
                    musicBtnText.innerText = "Pause Music";
                    musicIcon.innerText = "⏸️";
                }).catch(e => console.log("Audio play blocked", e));
            } else {
                music.pause();
                musicBtnText.innerText = "Play Music";
                musicIcon.innerText = "🎵";
            }
        }

        function typeWriter() {
            if (charIndex < quote.length) {
                document.getElementById("typingText").innerHTML += quote.charAt(charIndex);
                charIndex++;
                setTimeout(typeWriter, typingSpeed);
            } else {
                document.getElementById("typingText").style.borderRight = "none";
                setTimeout(transitionToPhotos, 2500);
            }
        }

        function transitionToPhotos() {
            document.getElementById("typingContainer").style.transition = "opacity 1s";
            document.getElementById("typingContainer").style.opacity = 0;

            setTimeout(() => {
                document.getElementById("typingContainer").style.display = "none";
                const photoBox = document.getElementById("photoContainer");
                photoBox.classList.add("show");
                startSlideshow();
            }, 1000);
        }

        let currentSlide = 0;
        const slides = document.querySelectorAll(".slide-img");
        const slideInterval = 3000;
        const bgGlow = document.getElementById("bgGlow");

        const colors = [
            "radial-gradient(circle, rgba(255, 51, 102, 0.25) 0%, rgba(0, 0, 0, 0) 70%)",
            "radial-gradient(circle, rgba(0, 255, 204, 0.2) 0%, rgba(0, 0, 0, 0) 70%)",
            "radial-gradient(circle, rgba(155, 89, 182, 0.23) 0%, rgba(0, 0, 0, 0) 70%)",
            "radial-gradient(circle, rgba(241, 196, 15, 0.18) 0%, rgba(0, 0, 0, 0) 70%)"
        ];

        function startSlideshow() {
            const timer = setInterval(() => {
                slides[currentSlide].classList.remove("active");
                currentSlide = currentSlide + 1;
                bgGlow.style.background = colors[currentSlide % colors.length];

                if (currentSlide >= slides.length) {
                    clearInterval(timer);
                    setTimeout(transitionToWish, 1000);
                } else {
                    slides[currentSlide].classList.add("active");
                }
            }, slideInterval);
        }

        function transitionToWish() {
            const photoBox = document.getElementById("photoContainer");
            photoBox.style.transition = "opacity 1s";
            photoBox.style.opacity = 0;
            bgGlow.style.background = "radial-gradient(circle, rgba(255, 51, 102, 0.35) 0%, rgba(0, 0, 0, 0) 70%)";

            setTimeout(() => {
                photoBox.style.display = "none";
                document.getElementById("wishContainer").classList.add("show");
                initHearts();
            }, 1000);
        }

        // Letter Actions
        function openLetter() {
            const overlay = document.getElementById("letterOverlay");
            overlay.style.display = "flex";
            setTimeout(() => overlay.classList.add("show"), 10);
        }

        function closeLetter() {
            const overlay = document.getElementById("letterOverlay");
            overlay.classList.remove("show");
            setTimeout(() => overlay.style.display = "none", 500);
        }

        // --- Canvas Hearts Animation ---
        const canvas = document.getElementById("heartCanvas");
        const ctx = canvas.getContext("2d");

        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        const hearts = [];

        class Heart {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = canvas.height + Math.random() * 100;
                this.size = Math.random() * 15 + 5;
                this.speedY = -(Math.random() * 1.5 + 0.5);
                this.speedX = Math.sin(Math.random() * 2) * 0.5;
                this.opacity = Math.random() * 0.7 + 0.3;
            }

            draw() {
                ctx.save();
                ctx.globalAlpha = this.opacity;
                ctx.fillStyle = "#ff3366";
                ctx.beginPath();
                ctx.moveTo(this.x, this.y + this.size / 4);
                ctx.quadraticCurveTo(this.x, this.y, this.x + this.size / 2, this.y);
                ctx.quadraticCurveTo(this.x + this.size, this.y, this.x + this.size, this.y + this.size / 3);
                ctx.quadraticCurveTo(this.x + this.size, this.y + (this.size * 2) / 3, this.x + this.size / 2, this.y + this.size);
                ctx.quadraticCurveTo(this.x, this.y + (this.size * 2) / 3, this.x, this.y + this.size / 3);
                ctx.quadraticCurveTo(this.x, this.y, this.x, this.y + this.size / 4);
                ctx.closePath();
                ctx.fill();
                ctx.restore();
            }

            update() {
                this.y += this.speedY;
                this.x += this.speedX;
                if (this.opacity > 0.01) this.opacity -= 0.001;
            }
        }

        function initHearts() {
            setInterval(() => {
                if (hearts.length < 80) hearts.push(new Heart());
            }, 150);
            animate();
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            for (let i = 0; i < hearts.length; i++) {
                hearts[i].update();
                hearts[i].draw();
                if (hearts[i].y < -20 || hearts[i].opacity <= 0.01) {
                    hearts.splice(i, 1);
                    i--;
                }
            }
            requestAnimationFrame(animate);
        }

        window.onload = typeWriter;
    </script>
